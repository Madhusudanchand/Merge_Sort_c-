//merge sort implementation
#include <iostream>

using namespace std;
void mergesort(int *ar,int lb,int ub);
void merge(int *ar,int lb,int mid,int ub);

int b[9];
int main()
{
	int ar[]={1,2,4,5,6,7,8,3,9};
	mergesort(ar,0,8);
	for(int i=0;i<9;i++)
	cout<<ar[i]<<" ";
	return 0;
}

void mergesort(int *ar,int lb,int ub)
{
	int mid;
	int i=lb;
	int j=ub;
	if(i<j)
	{
		mid=(i+j)/2;
		mergesort(ar,i,mid);
		mergesort(ar,mid+1,j);
		merge(ar,i,mid,j);
	}
}

void merge(int *a,int lb,int mid,int ub)
{
	int i=lb;
	int j=mid+1;
	int k=lb;
	while(i<=mid&&j<=ub)
	{
		if(a[i]<=a[j])
		{
			b[k]=a[i];
			i++;
		}
		else
		{
			b[k]=a[j];
			j++;
		}
		k++;
	}
	if(i>mid)
	{
		while(j<=ub)
		{
			b[k]=a[j];
			j++;
			k++;
		}
	}
	else
	{
		while(i<=mid)
		{
			b[k]=a[i];
			i++;
			k++;
		}
	}
	for(k=lb;k<=ub;k++)
	a[k]=b[k];
	
}
